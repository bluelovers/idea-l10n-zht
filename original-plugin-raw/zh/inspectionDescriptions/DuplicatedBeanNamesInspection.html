报告在应用程序上下文中定义的重复 Bean 名称。

示例：


  <beans profile="PROD">
    <bean class="..." id="duplicatedName"/>  <!-- 错误 -->
    <bean class="..." name="foo_name, duplicatedName"/> <!-- 'duplicatedName' 错误 -->
    <bean class="..." id="toBeAliased"/>

    <alias alias="duplicatedName" name="toBeAliased"/> <!-- 'duplicatedName' 错误 -->
  </beans>
