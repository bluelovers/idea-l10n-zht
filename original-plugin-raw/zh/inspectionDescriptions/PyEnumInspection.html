报告 Enum 的无效定义和用法。

示例： 


from enum import Enum


class Shape(Enum):
    SQUARE = 1
    CIRCLE = 2


class ExtendedShape(Shape):  # 枚举类 'Shape' 为 final，无法被子类化
    TRIANGLE = 3


  from enum import Enum


  class Color(Enum):
    _value_: int
    RED = 1
    GREEN = "green"  # 类型 'str' 不可赋值给已声明的类型 'int'


  from enum import Enum


  class Pet(Enum):
    CAT = 1
    DOG: int = 2  # 不允许为枚举成员使用类型注解
