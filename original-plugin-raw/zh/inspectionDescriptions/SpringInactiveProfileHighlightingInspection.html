报告 Spring XML 应用程序上下文中的停用配置文件。

示例：


  <!-- 为此应用程序上下文选择有效配置文件 'DEV' -->
  <beans profile="DEV">
    <bean class="MyFactory" factory-method="createBean"/>
  </beans>
  <!--  此标记的内容高亮显示为 'unused' -->
  <beans profile="TESTS">
    <bean class="MyTestFactory" factory-method="createBean"/>
  </beans>
  <!--  'unused' 部分的结尾 -->
  </beans>
