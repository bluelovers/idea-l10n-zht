報告靜態已知始終為 true、false、null 或零的非普通條件和值。 雖然有時會有意為之，但通常這是程序中存在邏輯錯誤的跡象。 此外，報告永遠無法到達 when 分支和某些靜態已知始終失敗的表達式。

範例：


fun process(x: Int?) {
  val isNull = x == null
  if (!isNull) {
    if (x != null) {} // 條件始終為 true
    require(x!! < 0 && x > 10) // 條件始終為 false
  } else {
    println(x!!) // !! operator will always fail
  }
}
fun process(v: Any) {
  when(v) {
    is CharSequence -> println(v as Int) // 轉換將始終失敗
    is String -> println(v) // 分支無法到達
  }
}
