報告不正確的切面和切入點。 當某些 AOP 建議與切面 Bean 匹配時，可能會導致執行時錯誤。

範例： 


<beans >
  <aop:config>
    <aop:pointcut expression="args()"/> <!-- 切入點應具有 ID -->
    <aop:aspect/> <!-- 應定義 'ref' 特性 -->
  </aop:config>
</beans>

<beans>
  <bean id="aspect1" class="Aspect1"/>
  <bean id="aspect2" class="Aspect2"/>
  <aop:config>
    <aop:aspect ref="aspect1"> <!-- 某些建議與此 Bean 符合 -->
    <aop:before method="xxx" pointcut="execution(* foo())"/>
    </aop:aspect>
    <aop:aspect ref="aspect2">
      <aop:before method="xxx" pointcut="execution(* foo())"/>
    </aop:aspect>
  </aop:config>
</beans>
