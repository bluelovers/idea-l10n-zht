報告 Spring Batch 基於 XML 的應用程式上下文的問題：
  • 未解析的 Bean 參照
  • 缺少標記和特性
  • 屬性類型不正確
  • enum 屬性不一致
  • 參照 Bean 的類型不正確

範例：


  <beans ... >
  <batch:job id="" <!-- Value must not be empty -->
    incrementer="dummyBean"  <!-- Bean 必須為 'org.springframework.batch.core.JobParametersIncrementer' 類型 -->
    job-repository="transactionManager"  <!-- Bean 必須為 'org.springframework.batch.core.repository.JobRepository' 類型 -->
    parent="INVALID_VALUE" <!-- 無法解析作業 INVALID_VALUE -->
    restartable="INVALID_VALUE"> <!-- 無法解析符號 INVALID_VALUE -->
    <batch:validator
      ref="dataSource">  <!-- Bean 必須為 'org.springframework.batch.core.JobParametersValidator' 類型 -->
    </batch:validator>
  </batch:job>
  <batch:job-repository/>

  <bean id="dataSource" class="org.springframework.jdbc.datasource.DelegatingDataSource"/>
  <bean id="transactionManager" class="org.springframework.jdbc.datasource.DataSourceTransactionManager"/>

  <bean id="dummyBean" class="java.lang.String"/>
</beans>
