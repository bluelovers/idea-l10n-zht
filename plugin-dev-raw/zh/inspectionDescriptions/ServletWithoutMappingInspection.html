報告沒有關聯映射的 servlet。

範例：


  <web-app>
    <servlet>
      <servlet-name>foo</servlet-name> // 錯誤：servlet 應該有一個映射
      <servlet-class>javax.servlet.Servlet</servlet-class>
    </servlet>
  
    <filter>
      <filter-name>bar</filter-name> // 錯誤：篩選器應該有一個映射
      <filter-class>java.lang.String</filter-class>
    </filter>
  </web-app>
