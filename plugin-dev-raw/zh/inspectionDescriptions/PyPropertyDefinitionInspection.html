報告 property() 的實參和用 @property 註解的函數的問題。

class C:
    @property
    def abc(self):  # Getter 應返回或生成某些內容
        pass

    @abc.setter
    def foo(self, value):  # 函數名稱與裝飾器不符合
        pass

    @abc.setter
    def abc(self, v1, v2):  # Setter 簽名應為 (self, value)
        pass

    @abc.deleter
    def abc(self, v1):  # Deleter 簽名應為 (self)
        pass
