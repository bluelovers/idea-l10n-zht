報告在 XML 應用程式上下文中定義的不正確架構 Bean。


  <beans ... xmlns:util="https://www.springframework.org/schema/util">
    <bean class="MyBean" id="myBean"/>
    <bean class="ErrTypeBean" id="errTypeBean"/>
    <util:list list-class="java.lang.String"/> // 報告 "類別必須為 'java.util.List' 的繼承者"
    <util:set value-type="MyBean">
      <ref bean="fooBean"/>
      <ref bean="errTypeBean"/> // 報告 "無法新增到 'MyBean' 類型的集合中"
    </util:set>
  </beans>
