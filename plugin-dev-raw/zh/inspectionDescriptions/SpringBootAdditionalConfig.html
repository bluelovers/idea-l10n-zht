報告 additional-spring-configuration-metadata.json 設定檔案中缺失和棄用的屬性、未解析的引用和無效值。

範例:


  {
    "properties": [
      {
        "name": "old.key",
        "type": "com.Unknown", // 報告 ‘無法解析類 ’Unknown‘”
        "description": "Description for old.key", // 報告"文本應以 '.' 結尾"
        "deprecation": {
          "replacement": "new.key", // Reports “無法解析設定鍵引用 'new.key‘”
          "reason": "Reason", // 報告“文本應以 '.' 結尾”
          "level": "warning"
         }
      },
      {
        "name": "duplicate", // 報告 "'duplicate' 的重複條目"
        "type": "java.lang.String"
      },
      {
        "name": "duplicate", // 報告 "'duplicate' 的重複條目"
        "type": "java.lang.String"
      },
      {
        // 報告 “缺失必要的 'name'"
        "type": "java.lang.String"
      }
    ]
  }
