報告 Enum 的無效定義和用法。

範例： 


from enum import Enum


class Shape(Enum):
    SQUARE = 1
    CIRCLE = 2


class ExtendedShape(Shape):  # 枚舉類別 'Shape' 為 final，無法被子類別化
    TRIANGLE = 3


  from enum import Enum


  class Color(Enum):
    _value_: int
    RED = 1
    GREEN = "green"  # 類型 'str' 不可賦值給已宣告的類型 'int'


  from enum import Enum


  class Pet(Enum):
    CAT = 1
    DOG: int = 2  # 不允許為枚舉成員使用類型註解
