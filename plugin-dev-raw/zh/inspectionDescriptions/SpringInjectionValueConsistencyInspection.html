報告 XML 應用程式上下文中注入點的不一致標記和特性。

  • <property> 和 <constructor-arg> 必須指定 ref、value 或子元素
  • <property> 和 <constructor-arg> 不能包含多個子元素
  • <ref> 可以包含 bean 特性、local 特性或 parent 特性
  • <ref> 必須指定一個 Bean
  • <idref> 可以包含 bean 特性或 local 特性
  • <idref> 必須指定一個 Bean
  • <map> 必須指定一個鍵
  • <idref> 可以包含 key 特性、key-ref 特性或 <key> 子元素

範例：


  <beans>

  <!-- 1.  屬性只允許包含 "ref" 特性或 "value" 特性或子元素 -->
  <bean class="MyClass">
    <property name="bean" ref="myBean" value="foo"/>
    // 屬性只允許包含 "ref" 特性或 "value" 特性或子元素
  </bean>

  <!--  2.  屬性不得包含多個子元素 -->
  <bean class="MyFactory">
    <property name="bean">
    <ref>bean</ref> // 屬性不得包含多個子元素
     <value>foo</value> // 屬性不得包含多個子元素
     </property>
  </bean>

  <!-- 3. 映射條目必須指定一個鍵  -->
  <bean class="MyBean">
     <property name="map">
        <map>
           <entry value="foo"/>  // 映射條目必須指定一個鍵
        </map>
     </property>
  </bean>
  </beans>
