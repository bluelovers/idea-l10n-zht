報告 Spring XML 應用程式上下文中的非活動設定檔案。

範例:


  <!-- 為該套用程序上下文選擇了活動設定檔案 'DEV' -->
  <beans profile="DEV">
    <bean class="MyFactory" factory-method="createBean"/>
  </beans>
  <!--  此標籤的內容高亮顯示為 ‘unused’ -->
  <beans profile="TESTS">
    <bean class="MyTestFactory" factory-method="createBean"/>
  </beans>
  <!--  ‘unused’ 部分的結尾 -->
  </beans>
