報告給方法接收器的賦值。

當您將值賦給方法接收器時，該值將不會反映在方法本身之外。 值將反映在來自同一方法的後續調用中。

範例：

package main

import "fmt"

type demo struct {
 Val int
}

func (d *demo) change() {
 d = nil // 對方法接收器的分配僅傳播給被調用者，而不傳播給調用者
 d.myVal()
}

func (d *demo) myVal() {
 fmt.Printf("my val: %#v\n", d)
}

func (d demo) change2() {
 d = demo{} // 對方法接收器的分配不會傳播到其他調用
 d.myVal()
}

func (d *demo) change3() {
 d.Val = 3
 d.myVal()
}

func main() {
 d := &demo{}
 d.myVal()
 d.change()
 d.myVal()
 d.Val = 2
 d.change2()
 d.myVal()
 d.change3()
 d.myVal()
}
