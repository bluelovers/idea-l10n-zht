  // 1.  @PayloadRoot on non-endpoint classes

	@PayloadRoot(namespace = NAMESPACE_URI, localPart = "HolidayRequest") // @Endpoint 類支援 @PayloadRoot

     ...
  }

  // 2.  @PayLoad 方法上的 @XPathParam
  @Endpoint
  public class AirlineEndpoint {
	  @Namespace(prefix = "m", uri = MESSAGES_NAMESPACE) @ResponsePayload
	  public GetFlightsResponse getFlights(@XPathParam("//m:sc") String serviceClassString) {..} //@PayLoad 方法支援 @XPathParam
  }
}
