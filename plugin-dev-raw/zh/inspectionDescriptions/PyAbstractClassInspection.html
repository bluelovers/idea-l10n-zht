報告無效的抽象類別定義和用法。

範例：


from abc import abstractmethod, ABC


class Figure(ABC):

    @abstractmethod
    def do_figure(self):
        pass


class Triangle(Figure):  # 'Triangle' 類別中未定義所有 abstract 方法
    def do_triangle(self):
        pass


Triangle()  # 無法實例化抽象類別 'Triangle'

套用快速修復後，IDE 為 Triangle 類別實作抽象方法：


from abc import abstractmethod, ABC


class Figure(ABC):

    @abstractmethod
    def do_figure(self):
        pass


class Triangle(Figure):
    def do_figure(self):
        pass

    def do_triangle(self):
        pass


Triangle()
